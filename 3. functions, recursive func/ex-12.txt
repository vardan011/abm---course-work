//swap function
#include <iostream>
using namespace std;

void Swap(int&, int&);
 
int main() {
    int a, b;
    cout << "a = "; cin >> a;
    cout << "b = "; cin >> b;
    Swap(a, b);
    cout << "Значения переменных после обмена:\n"
         << "a = " << a << endl
         << "b = " << b << endl;
    return 0;
}
 
void Swap(int& x, int& y) {
    if (x == y)
        return;
    x ^= y;
    y ^= x;
    x ^= y;
}
